package com.example.wifisuite

import android.content.Context
import android.content.ClipData
import android.content.ClipboardManager
import android.content.Intent
import android.net.wifi.WifiManager
import android.os.Bundle
import android.widget.Toast
import androidx.activity.ComponentActivity
import androidx.activity.compose.setContent
import androidx.compose.foundation.layout.*
import androidx.compose.foundation.lazy.LazyColumn
import androidx.compose.material.*
import androidx.compose.runtime.*
import androidx.compose.ui.Modifier
import androidx.compose.ui.unit.dp

data class WiFiInfo(
    val ssid: String,
    val bssid: String,
    val ip: String,
    val signal: Int,
    val speed: Int,
    val timestamp: String
) {
    fun prettyString() = "SSID: $ssid\nBSSID: $bssid\nIP: $ip\nSignal: $signal dBm\nSpeed: $speed Mbps\nTime: $timestamp"
}

class MainActivity : ComponentActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContent { WiFiSuiteApp(this) }
    }
}

@Composable
fun WiFiSuiteApp(context: Context) {
    var networkInfo by remember { mutableStateOf(getWifiInfo(context)) }
    var showQR by remember { mutableStateOf(false) }
    var history by remember { mutableStateOf(listOf(networkInfo)) }
    var cloudSync by remember { mutableStateOf(true) }

    MaterialTheme {
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(24.dp)
        ) {
            Row(Modifier.fillMaxWidth(), horizontalArrangement = Arrangement.SpaceBetween) {
                Text("WiFiSuite", style = MaterialTheme.typography.h5)
                Button(onClick = {
                    // Stub: handle logout
                }) {
                    Text("Logout")
                }
            }
            Spacer(Modifier.height(8.dp))
            Text("Welcome, User!")
            Spacer(Modifier.height(16.dp))
            Card(
                elevation = 6.dp,
                modifier = Modifier.fillMaxWidth()
            ) {
                Column(Modifier.padding(16.dp)) {
                    Text("SSID: ${networkInfo.ssid}")
                    Text("BSSID: ${networkInfo.bssid}")
                    Text("IP: ${networkInfo.ip}")
                    Text("Signal: ${networkInfo.signal} dBm")
                    Text("Speed: ${networkInfo.speed} Mbps")
                }
            }
            Spacer(Modifier.height(12.dp))
            Row {
                Button(onClick = {
                    val clipboard = context.getSystemService(Context.CLIPBOARD_SERVICE) as ClipboardManager
                    clipboard.setPrimaryClip(ClipData.newPlainText("Wi-Fi Info", networkInfo.prettyString()))
                    Toast.makeText(context, "Copied!", Toast.LENGTH_SHORT).show()
                }) { Text("Copy") }
                Spacer(Modifier.width(8.dp))
                Button(onClick = {
                    val sendIntent = Intent().apply {
                        action = Intent.ACTION_SEND
                        putExtra(Intent.EXTRA_TEXT, networkInfo.prettyString())
                        type = "text/plain"
                    }
                    context.startActivity(Intent.createChooser(sendIntent, "Share Wi-Fi Info"))
                }) { Text("Share") }
                Spacer(Modifier.width(8.dp))
                Button(onClick = { showQR = true }) { Text("Show QR") }
                Spacer(Modifier.width(8.dp))
                Button(onClick = {
                    history = listOf(networkInfo) + history
                    Toast.makeText(context, "Saved!", Toast.LENGTH_SHORT).show()
                }) { Text("Save/Sync") }
            }
            // (Omit QR modal for brevity; see previous responses for QR logic)
            Spacer(Modifier.height(16.dp))
            Text("History", style = MaterialTheme.typography.h6)
            LazyColumn(modifier = Modifier.weight(1f)) {
                items(history.size) { idx ->
                    Text("${history[idx].ssid} – ${history[idx].timestamp}")
                }
            }
            Row(Modifier.fillMaxWidth(), horizontalArrangement = Arrangement.SpaceBetween) {
                Row(verticalAlignment = Alignment.CenterVertically) {
                    Switch(checked = cloudSync, onCheckedChange = { cloudSync = it })
                    Text("Cloud Sync: ${if (cloudSync) "ON" else "OFF"}")
                }
                // Logout handled above
            }
        }
    }
}

fun getWifiInfo(context: Context): WiFiInfo {
    val wifiManager = context.applicationContext.getSystemService(Context.WIFI_SERVICE) as WifiManager
    val info = wifiManager.connectionInfo
    return WiFiInfo(
        ssid = info.ssid,
        bssid = info.bssid,
        ip = android.text.format.Formatter.formatIpAddress(info.ipAddress),
        signal = info.rssi,
        speed = info.linkSpeed,
        timestamp = System.currentTimeMillis().toString()
    )
}
