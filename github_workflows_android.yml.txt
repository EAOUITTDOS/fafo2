name: Android CI

on:
  push:
    paths:
      - 'android/**'
      - '.github/workflows/android.yml'
  pull_request:
    paths:
      - 'android/**'

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
    - name: Checkout repo
      uses: actions/checkout@v4
    - name: Set up JDK 17
      uses: actions/setup-java@v4
      with:
        distribution: 'temurin'
        java-version: 17
    - name: Build with Gradle
      run: cd android && ./gradlew build
